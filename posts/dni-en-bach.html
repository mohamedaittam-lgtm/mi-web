@echo off

set /p dni= "Introduce un NIF: "

set cadena=TRWAGMYFPDXBNJZSQVHLCKE

set /a resultado=%dni%-(23*(%dni%/23))

:bucle

if %resultado%==0 goto fin

::resto --

set /a resultado-=1

set cadena=%cadena:~1,23%

::echo %cadena%

goto bucle

:fin

echo La letra del NIF es %cadena:~0,1%

pause>nul
