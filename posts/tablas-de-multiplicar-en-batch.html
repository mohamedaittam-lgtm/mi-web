@echo off

color 30

setlocal enabledelayedexpansion

cls

set /p a="Introduzca un numero:"

echo ================================

echo = TABLA DE MULTIPLICAR DE %a%   =

echo ================================

echo.

for  /L %%b in (1,1,10) do (

  set /a resultado=%a% * %%b

  echo %a% * %%b = !resultado!

  )

pause
